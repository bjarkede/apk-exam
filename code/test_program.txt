// Testing higher-order functions
// Where we pass a a function as an argument to another function.
let f(x) = 
	let g(y) = x * y in g end 
	in print(f(6) (7))
end

// Testing fib and compile-time things
let input = 10 end

// keyword using compile-time array generation
print(fib(input))

// run-time implementation
// higher than runtimefib(31) causes bad_alloc...
let runtimefib(n) = 
	if n < 2 then n else runtimefib(n - 1) + runtimefib(n - 2) 
	in print(runtimefib(input)) 
end
